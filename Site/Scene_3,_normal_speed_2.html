% Examples of plot



% Plot pedestrians with dots

% here after 500 time steps

n_time = 500;

f = population.pop_display('plot','positions', environment, n_time);





















% Plot pedestrians with the corresponding index

% (useful when the density is low)

% here after 90 time steps

n_time = 90;

f = population.pop_display('plot-idx','positions', environment, n_time);






















% Plot the continuous density

% here after 900 time steps

n_time = 900;

f = population.pop_display('density','positions', environment, n_time);






















% Plot the dense velocity field

% (vector fields with norm of the velocity)

% here after 6500 time steps

n_time = 650;

f = population.pop_display('continuum','velocities', environment_r2, n_time);




















% Export the current figure

export_fig('velocities.pdf');
