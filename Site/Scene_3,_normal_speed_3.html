% Global variables related to displays

Ntimes = 999999;

% Ntimes is the maximal time of the sequence

% (if superior to the actual sequence, then the

% complete one is treated)


% Generate images for a movie corresponding to the density

% with one image every 80

frames_modulo = 80;

M = population.pop_display('movie-density','positions', environment_r2, Ntimes, frames_modulo, 'movie-density-positions');

% Snapshot























% Generate images for a movie corresponding to vector fields

% with one image every 80

frames_modulo = 80;

M = population.pop_display('movie-continuum','velocities', environment_r2, Ntimes, frames_modulo, 'movie-continuum-velocities');

% Snapshot























% Generate images for a movie corresponding to circles

% linked to the radius of pedestrians

% with one image every 100

frames_modulo = 100;

% Snapshot























% Generate images for a movie corresponding to circles
